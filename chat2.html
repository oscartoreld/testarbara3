  <script>

    async function fetchLiveData() {

      const res = await fetch('https://site.web.api.espn.com/apis/site/v2/sports/golf/pga/leaderboard');

      const data = await res.json();

      const players = data.events[0].competitions[0].competitors;


      const leaderboardData = players.map(p => {

        const name = p.athlete.displayName;

        const rounds = p.linescores.map(r => parseInt(r.displayValue) || 0);

        const total = rounds.reduce((a, b) => a + b, 0);

        return { name, rounds, total };

      }).sort((a, b) => a.total - b.total);


      const tbody = document.querySelector("#leaderboard tbody");

      tbody.innerHTML = "";

      leaderboardData.forEach((p, i) => {

        tbody.innerHTML += `

          <tr>

            <td>${i + 1}</td>

            <td>${p.name}</td>

            <td>${p.rounds[0] ?? ''}</td>

            <td>${p.rounds[1] ?? ''}</td>

            <td>${p.rounds[2] ?? ''}</td>

            <td>${p.rounds[3] ?? ''}</td>

            <td><strong>${p.total}</strong></td>

          </tr>

        `;

      });

    }


    fetchLiveData();

    setInterval(fetchLiveData, 10000);

  </script>
